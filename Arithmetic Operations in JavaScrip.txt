/* Arithmetic Operations in JavaScript */

let a=5;
let b=2;

console.log("a=",a, "& b=",b);
console.log("a + b = ",a+b);
console.log("a - b = ",a-b);
console.log("a * b = ",a*b);
console.log("a/b = ",a/b);
console.log("a % b = ",a%b); /*This is "Modulus" operator*/
console.log("a ** b = ",a**b); /*this code basically refers a to the power b*/

/* Unary Operator */
/*Unary Operators basically refers the increment or decremnt*/

a++; /*you must have to write this line, rather code will not work*/
console.log("a++ = ",a);

b--;
console.log("b-- = ",b);

/* Post Incremnt */
/*post increment basically ja kore ta holo, at first post increment nijer first value ta print koreh, then next line e value increase kore*/
let f=3;
console.log("f++ = ",f++);
console.log("f++ = ",f++);
console.log("f++ = ",f++);

/* Pre Increment */
/*Pre Increment basically ja kore ta holo, at first tar value ke increase kore then print kore*/
let g=9;
console.log("++g = ",++g);
console.log("++g= ",++g);

/* Post Decrement */
/*at first post decrement nijer fisrt given value print koreh, then next line e ta decrease kore*/
let m=10;
console.log("m-- = ",m--);
console.log("m-- = ",m--);
console.log("m-- = ",m--);

/* Pre Decrement */
/*Pre decrement at first value decrease korbe, then ta print korbeh*/
let h=12;
